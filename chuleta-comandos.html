Chuleta de Comandos

YT-DLP

# Descargar MP4
yt-dlp -f "bestvideo[ext=mp4]+bestaudio[ext=m4a]/mp4" URL

# Descargar MP3 con metadatos
yt-dlp -x --audio-format mp3 --audio-quality 0 --embed-metadata --embed-thumbnail --add-metadata --extractor-args "youtube:player_client=android" -o "%(artist,creator,uploader)s - %(title)s.%(ext)s" URL

FFmpeg

# Capturar pantalla
ffmpeg -f x11grab -framerate 30 -i :0.0 output.mp4
# Presionar Q para detener

# Video a GIF
ffmpeg -i input.mp4 -vf "fps=15,scale=600:-1:flags=lanczos" output.gif

# Miniaturas escaladas
ffmpeg -i %d.jpg -vf "scale=320:-1" %d_thumb.jpg

Aria2

# Instalar
sudo pacman -S aria2

# Descargar archivo
aria2c https://ejemplo.com/archivo.zip

# Reanudar descarga interrumpida
aria2c -c https://ejemplo.com/archivo.zip

MPV

# Stream de Twitch
mpv https://www.twitch.tv/usuario

# Solo audio de Twitch
mpv --no-video https://www.twitch.tv/usuario

# YouTube en bucle (solo audio)
mpv --no-video --loop https://youtube.com/watch?v=ejemplo

# Detener: Ctrl + C

Find

# Buscar archivos por nombre
find /ruta -name "archivo.txt"

# Buscar por extensión
find /ruta -name "*.log"

# Buscar archivos modificados en los últimos 7 días
find /ruta -mtime -7

# Buscar directorios
find /ruta -type d -name "carpeta"

# Buscar y eliminar archivos mayores a 100MB
find /ruta -type f -size +100M -exec rm {} \;

# Buscar archivos por contenido (encontrar txt que contenga "palabra")
find /ruta -name "*.txt" -exec grep -l "palabra" {} \;

Grep

# Buscar palabra en archivo
grep "palabra" archivo.txt

# Buscar recursivamente en directorio
grep -r "palabra" /directorio

# Mostrar número de línea
grep -n "palabra" archivo.txt

# Buscar sin distinción mayúsculas/minúsculas
grep -i "palabra" archivo.txt

# Excluir palabras
grep -v "excluir" archivo.txt

# Expresiones regulares
grep -E "patron[0-9]+" archivo.txt

Nano + FZF

# Instalar FZF
sudo pacman -S fzf

# Buscar y editar archivo
nano $(fzf)

# Alternar preview en FZF
fzf --preview 'cat {}'

# Buscar en archivos con grep y abrir con nano
grep -r "palabra" . | fzf | xargs nano

Combinaciones Útiles

# Encontrar y editar
find . -name "*.md" | fzf | xargs nano

# Buscar texto en archivos y editar resultado
grep -r "error" . | fzf -m | xargs -r nano

# Editar último archivo modificado
nano $(ls -t *.txt | head -1)

# Ver contenido sin abrir editor
find . -name "*.log" -exec tail -n 20 {} \;

# Matar proceso por nombre
pkill -f "proceso"
