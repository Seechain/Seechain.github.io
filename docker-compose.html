Docker Compose - Servicios

Servicios configurados usando Docker Compose Generator:

  • Jellyfin - Media server
  • Heimdall - Dashboard de aplicaciones
  • Filebrowser - Gestor de archivos
  • Transmission - Cliente BitTorrent

Archivo docker-compose.yml

services:

  jellyfin:
    image: lscr.io/linuxserver/jellyfin:latest
    container_name: jellyfin
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=America/Santiago
    volumes:
      - /home/tu_usuario/docker-services/jellyfin/config:/config
      - /home/tu_usuario/peliculas:/data/movies
      - /home/tu_usuario/musica:/data/music
      - /home/tu_usuario/series:/data/series
    ports:
      - "8096:8096"
      - "7359:7359/udp"
    restart: unless-stopped

  heimdall:
    image: lscr.io/linuxserver/heimdall:latest
    container_name: heimdall
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=America/Santiago
    volumes:
      - /home/tu_usuario/docker-services/heimdall/config:/config
    ports:
      - "80:80"
      - "443:443"
    restart: unless-stopped

  filebrowser:
    image: filebrowser/filebrowser
    container_name: filebrowser
    user: "1000:1000"
    ports:
      - "8081:80"
    volumes:
      - /home/tu_usuario:/srv
      - /home/tu_usuario/docker-services/filebrowser/config:/config
    restart: unless-stopped

  transmission:
    image: linuxserver/transmission:latest
    container_name: transmission
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=America/Santiago
      - TRANSMISSION_WATCH_DIR_ENABLED=true
      - TRANSMISSION_WATCH_DIR=/watch
    volumes:
      - /home/tu_usuario/docker-services/transmission/config:/config
      - /home/tu_usuario/descargas:/downloads
      - /home/tu_usuario/docker-services/transmission/watch:/watch
    ports:
      - "9091:9091"
      - "51413:51413" 
      - "51413:51413/udp"
    restart: unless-stopped

Uso

# Iniciar servicios
docker compose up -d

# Ver estado
docker compose ps

# Ver logs
docker compose logs -f
