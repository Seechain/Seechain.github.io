Comandos Docker

Información

# Información del sistema Docker (versión, contenedores, recursos)	
sudo docker info 

# Estado del servicio Docker (activo/inactivo/errores)
systemctl status docker  

# Lista todos los contenedores (activos y detenidos)
docker ps -a    

# Lista todas las imágenes descargadas         
docker images  
   
# Monitor de recursos en tiempo real (CPU, memoria, red)      
docker stats 

# Muestra los logs de un contenedor específico
sudo docker logs nombre-contenedor

Docker Compose

# Iniciar servicios
docker compose up -d

# Detener servicios
docker compose down

# Ver logs
docker compose logs -f

# Ver estado
docker compose ps

# Reconstruir servicios
docker compose build

# Reiniciar un servicio
docker compose restart nombre-servicio

# Validar YAML
docker compose config

Limpieza

# Eliminar contenedores parados, imágenes y redes sin usar
docker system prune -a

# Lo anterior + volúmenes sin usar
docker system prune -a --volumes

# Eliminar todos los contenedores
docker container prune -f

# Eliminar todas las imágenes
docker image prune -a

# Eliminar todos los volúmenes
docker volume prune -f

# Eliminar redes sin usar
docker network prune -f
